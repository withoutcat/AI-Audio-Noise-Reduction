200 Optional<bool> playbackAndRecord;

299 void SetAll(const AudioSessionConfiguration& change) {
300 SetFrom(&playbackAndRecord, change.playbackAndRecord);
301 SetFrom(&chatMode, change.chatMode);
302 SetFrom(&defaultToSpeaker, change.defaultToSpeaker);
303 SetFrom(&overrideSpeaker, change.overrideSpeaker);
304 SetFrom(&allowMixWithOthers, change.allowMixWithOthers);
305 SetFrom(&allowDuckOthers, change.allowDuckOthers);
306 SetFrom(&allowBluetooth, change.allowBluetooth);
307 SetFrom(&allowBluetoothA2DP, change.allowBluetoothA2DP);
308 SetFrom(&sampleRate, change.sampleRate);
309 SetFrom(&ioBufferDuration, change.ioBufferDuration);
310 SetFrom(&inputNumberOfChannels, change.inputNumberOfChannels);
311 SetFrom(&outputNumberOfChannels, change.outputNumberOfChannels);
312#if defined(WEBRTC_IOS)
313 SetFrom(&category, change.category);
314#endif
315 }
316
317 bool operator==(const AudioSessionConfiguration& o) const {
318 return playbackAndRecord == o.playbackAndRecord && chatMode == o.chatMode &&
319 defaultToSpeaker == o.defaultToSpeaker && overrideSpeaker == o.overrideSpeaker &&
320 allowMixWithOthers == o.allowMixWithOthers && allowDuckOthers == o.allowDuckOthers &&
321 allowBluetooth == o.allowBluetooth && allowBluetoothA2DP == o.allowBluetoothA2DP && sampleRate == o.sampleRate &&
322 ioBufferDuration == o.ioBufferDuration &&
323 inputNumberOfChannels == o.inputNumberOfChannels &&
324#if defined(WEBRTC_IOS)
325 outputNumberOfChannels == o.outputNumberOfChannels &&
326 category == o.category;
327#else
328 outputNumberOfChannels == o.outputNumberOfChannels;
329#endif
330 }
331 bool operator!=(const AudioSessionConfiguration& o) const { return !(*this == o); }
332
333 private:
334 template <typename T>
335 static void SetFrom(Optional<T>* s, const Optional<T>& o) {
336 if (o) {
337 *s = o;
338 }
