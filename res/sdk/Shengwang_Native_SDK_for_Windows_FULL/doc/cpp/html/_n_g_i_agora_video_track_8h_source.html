39struct StreamLayerConfigInternal {
40 VideoDimensions dimensions;
41 int framerate;
42 int bitrate_kbps;
43 bool enable;
44 StreamLayerConfigInternal() : dimensions(0, 0), framerate(0), bitrate_kbps(STANDARD_BITRATE), enable(false) {}
45 StreamLayerConfigInternal(const StreamLayerConfigInternal& other) : dimensions(other.dimensions), framerate(other.framerate), bitrate_kbps(other.bitrate_kbps), enable(other.enable) {}
46 bool operator==(const StreamLayerConfigInternal& rhs) const {
47 return dimensions == rhs.dimensions && bitrate_kbps == rhs.bitrate_kbps && framerate == rhs.framerate && enable == rhs.enable;
48 }
49
50 StreamLayerConfigInternal& operator=(const SimulcastConfig::StreamLayerConfig& slc) {
51 dimensions = slc.dimensions;
52 framerate = slc.framerate;
53 enable = slc.enable;
54 return *this;
55 }
56
57 void reset() {
58 dimensions.width = 0;
59 dimensions.height = 0;
60 framerate = 0;
61 bitrate_kbps = STANDARD_BITRATE;
62 enable = false;
63 }
64};

66struct SimulcastConfigInternal {
67 StreamLayerConfigInternal simulcastlayerConfigs[STREAM_LAYER_COUNT_MAX];
68
69 bool publish_fallback_enable;
70 bool publish_on_demand;
71
72 void reset() {
73 for (int i = STREAM_LAYER_1; i < STREAM_LAYER_COUNT_MAX; i++) {
74 simulcastlayerConfigs[i].reset();
75 }
76 }
77
78 bool operator==(const SimulcastConfigInternal& rhs) const {
79 for (int i = 0; i < STREAM_LAYER_COUNT_MAX; i++) {
80 if (simulcastlayerConfigs[i] == rhs.simulcastlayerConfigs[i]) {
81 continue;
82 }
83 return false;
84 }
85 return publish_fallback_enable == rhs.publish_fallback_enable &&
86 publish_on_demand == rhs.publish_on_demand;
87 }
88
89 SimulcastConfigInternal(): publish_fallback_enable(false), publish_on_demand(true) {}
90};
